040 /**
041  * Represents the metadata for a bean properties and actions. All beans must be Serializable by Wicket
042  * convention.<p/>
043  * 
044  * By default, the metadata originates by convention:<p/>
045  * <ul>
046  * <li>Label names for properties are derived from the JavaBean spec "displayName" or 
047  *     from the property name. E.g.,
048  *      "customerName" becomes "Customer Name"; "address2" becomes "Address 2".
049  * </li>
050  * <li>Field components for the Java primitive/wrapper types, enum types,
051  *      java.util.Date and those classes deriving from it, and java.util.Lists are pre-configured.
052  * </li>
053  * <li>All JavaBean properties are displayed.
054  * </li>
055  * <li>All fields are editable if viewOnly (see constructor) is false. 
056  *      Otherwise they are all view-only.
057  * </li>
058  * <li>If a property is not writable, it is displayed view-only.
059  * </li>
060  * <li>All fields are displayed in the order the are reflected.
061  * </li>
062  * <li>All fields are displayed in a single default tab.
063  * </li>
064  * </ul> 
065  * <p/>
066  * 
067  * The Field types for classes can be specified/overridden in the given ComponentRegistry.  
068  * <p/>
069  * 
070  * Normally, this class does what you would expect. 
071  * However, you can override the default conventions by specifying the exceptions in 
072  * the given Component's ".beanprops" file (see constructor). 
073  * <p/>
074  * 
075  * If no tabs are specified, a default tab with an id of "DEFAULT_TAB" and a label based on the bean class name. 
076  * Property-level settings override Bean-level settings. 
077  * <p/>
078  * 
079  * Certain components that use BeanMetaData listen to PropertyChangeEvents. If your bean implements
080  * addPropertyChangeListener/removePropertyChangeListener bean methods, the component can listen for 
081  * property changes and update components on the page dynamically.
082  * <p/>
083  *  
084  * @author Dan Syrstad
085  */
086 public class BeanMetaData extends MetaData implements Serializable

182     /**
183      * Determines if all parameters specified have been consumed for a specific tab, or all tabs.
184      * 
185      * @param unconsumedMsgs messages that report the parameter keys that were specified but not consumed.
186      * @param tabMetaData the tab to be checked. If null, all elements and tabs are checked.
187      * 
188      * @return true if all parameters specified have been consumed.
189      */
190     public boolean areAllParametersConsumed(Set<String> unconsumedMsgs, TabMetaData tabMetaData)
191     {
192         if (!super.areAllParametersConsumed("Bean " + beanClass.getName(), unconsumedMsgs)) {
193             return false;
194         }
195         
196         // Make sure all elements and tabs have their parameters consumed.
197         for (ElementMetaData element : tabMetaData == null ? getDisplayedElements() : getTabElements(tabMetaData)) {
198             if (!element.areAllParametersConsumed("Property " + element.getPropertyName(), unconsumedMsgs)) {
199                 return false;
200             }
201         }
202         
203         for (TabMetaData tab : tabMetaData == null ? tabs : Collections.singletonList(tabMetaData)) {
204             if (!tab.areAllParametersConsumed("Tab " + tab.getId(), unconsumedMsgs)) {
205                 return false;
206             }
207         }
208         
209         return true;
210     }

212     /**
213      * Logs a warning if any parameter specified have not been consumed for a specific tab, or all tabs.
214      * 
215      * @param tabMetaData the tab to be checked. If null, all elements and tabs are checked.
216      */
217     public void warnIfAnyParameterNotConsumed(TabMetaData tabMetaData)

221             for (String msg : msgs) {
222                 logger.warning(msg);
223             }
224         }
225     }

249     private void init()
250     {
251         // Check if bean supports PropertyChangeListeners.
252         hasAddPropertyChangeListenerMethod = getAddPropertyChangeListenerMethod() != null;
253         hasRemovePropertyChangeListenerMethod = getRemovePropertyChangeListenerMethod() != null;
254         
255         // Create defaults based on the bean itself.
256         PropertyDescriptor[] descriptors = PropertyUtils.getPropertyDescriptors(beanClass);
257         for (PropertyDescriptor descriptor : descriptors) {
258             String name = descriptor.getName();
259             
260             // Skip getClass() and methods that are not readable or hidden.
261             if (name.equals("class") || descriptor.getReadMethod() == null || descriptor.isHidden()) {
262                 continue;
263             }
264             
265             String label = descriptor.getDisplayName();
266             if (label.equals(name)) {
267                 label = createLabel(name);
268             }
269 
270             ElementMetaData propertyMeta = new ElementMetaData(this, name, label, descriptor.getPropertyType());
271             propertyMeta.setViewOnly( isViewOnly() );
272             elements.add(propertyMeta);
273 
274             if (descriptor.getWriteMethod() == null) {
275                 propertyMeta.setViewOnly(true);
276             }
277         }
278 
279         String propFileName = getBaseClassName(component.getClass()) + ".beanprops";
280         List<Bean> beanSpecs = cachedBeanProps.get(propFileName);
281         if (beanSpecs == null) {
282             // It's OK not to have a beanprops file. We can deduce the parameters by convention. 
283             InputStream propsStream = component.getClass().getResourceAsStream(propFileName);
284             if (propsStream != null) {
285                 try {
286                     beanSpecs = new BeanPropsParser(propFileName, propsStream).parse();
287                     cachedBeanProps.put(propFileName, beanSpecs);
288                 }
289                 finally {
290                     try { propsStream.close(); } catch (IOException e) { /* Ignore */ }
291                 }
292             }
293         }
294         
295         if (beanSpecs != null) {
296             processBeanSpecs(beanSpecs);
297         }
298 
299         // Process Bean-level parameters
300         if (!getBooleanParameter(PARAM_DISPLAYED)) {
301             elements.clear();
302             tabs.clear();
303         }
304         
305         Boolean beanViewOnly = (getParameter(PARAM_VIEW_ONLY) == null ? null : isViewOnly());
306         
307         // Configure tabs
308         if (tabs.isEmpty()) {
309             // Create single default tab.
310             tabs.add( new TabMetaData(this, DEFAULT_TAB_ID, getParameter(PARAM_LABEL) ) );
311         }
312         
313         String defaultTabId = tabs.get(0).getId();
314         
315         // Post-process each property based on bean parameters
316         for (ElementMetaData elementMeta : elements) {
317             // If bean specified view only, but element didn't, set it to the bean's value.
318             if (beanViewOnly != null && !elementMeta.isViewOnlySetExplicitly()) {
319                 // Don't override default view-only (based on bean) if bean is not viewOnly.
320                 if (!elementMeta.isViewOnly()) {
321                     elementMeta.setViewOnly(beanViewOnly);
322                 }
323             }
324             
325             // If element is not on a tab, add it to the first. If it's an action, it must have been assigned an order to
326             // appear on a tab. Otherwise it is a global action.
327             if (elementMeta.getTabId() ==  null &&
328                 (!elementMeta.isAction() || 
329                  (elementMeta.isAction() && elementMeta.getOrder() != ElementMetaData.DEFAULT_ORDER))) {
330                 elementMeta.setTabId(defaultTabId);
331             }
332         }
333 
334         Collections.sort(elements, new Comparator<ElementMetaData>() {
335             public int compare(ElementMetaData o1, ElementMetaData o2)
336             {
337                 return (o1.getOrder() > o2.getOrder() ? 1 : (o1.getOrder() < o2.getOrder() ? -1 : 0));
338             }
339         });
340     }

342     /**
343      * Gets the base class name of a Class.
344      * 
345      * @param aClass the class.
346      *
347      * @return the base class name (the name without the package name).
348      */
349     private static String getBaseClassName(Class aClass) 
350     {
351         String baseClassName = aClass.getName();
352         int idx = baseClassName.lastIndexOf('.');
353         if (idx >= 0) {
354             baseClassName = baseClassName.substring(idx + 1);
355         }
356 
357         return baseClassName;
358     }

360     /**
361      * Process bean ASTs that apply to this bean. 
362      *
363      * @param beans the Bean ASTs.
364      */
365     private void processBeanSpecs(List<Bean> beans)
366     {
367         // Determine the hierarchy of Bean ASTs. I.e., the default Bean is always processed first, followed by those that
368         // extend it, etc.
369         // This acts as a stack.
370         List<Bean> beansHier = new ArrayList<Bean>();
371         String currContext = context;
372         // Note: Limit cyclical specs (e.g., A extends B, B extends A). This also limits the maximum hierarchy depth to the same 
373         // amount, which should be plenty.
374         for (int limit = 0; limit < 20; ++limit) { 
375             Bean bean = getBean(beans, currContext);
376             beansHier.add(0, bean);
377             if (currContext == null) {
378                 // Just processed the default context, so stop.
379                 break;
380             }
381             
382             currContext = bean.getExtendsContext();
383         }
384         
385         // Apply beans in order from highest to lowest. The default context will always be first.
386         for (Bean bean : beansHier) {
387             applyBean(bean);
388         }
389     }

391     /**
392      * Applies a Bean AST to this meta data.
393      *
394      * @param bean
395      */
396     private void applyBean(Bean bean)
397     {
398         // Process actions first.
399         for (Parameter param :  bean.getParameters()) {
400             if (param.getName().equals(PARAM_ACTIONS)) {
401                 applyActions(param.getValues());
402             }
403         }
404         
405         for (Parameter param :  bean.getParameters()) {
406             String name = param.getName();
407             if (name.equals(PARAM_PROPS)) {
408                 applyProps(param.getValues(), null);
409             }
410             else if (name.equals(PARAM_TABS)) {
411                 // Ignore - processed below.
412             }
413             else if (name.equals(PARAM_ACTIONS)) {
414                 // Ignore - already processed above.
415             }
416             else {
417                 // Just handle a regular single-valued parameter.
418                 List<ParameterValue> values = param.getValues();
419                 if (values.size() != 1) {
420                     throw new RuntimeException("Parameter " + name + " on bean " + bean.getName() + " does not specify exactly one value.");
421                 }
422                 
423                 String value = values.get(0).getValue(component);
424                 setParameter(name, value);
425             }
426         }
427 
428         // Process tabs last.
429         for (Parameter param :  bean.getParameters()) {
430             if (param.getName().equals(PARAM_TABS)) {
431                 applyTabs(param.getValues());
432             }
433         }
434     }

436     /**
437      * Applies a Bean's "props" to each ElementMetaData.
438      *
439      * @param values
440      * @param tabId the tab id to apply to the properties. May be null, in which case the tab id is not affected.
441      */
442     void applyProps(List<ParameterValue> values, String tabId)
443     {
444         int order = 1;
445         for (ParameterValue value : values) {
446             String elementName = value.getValue();
447             boolean removeElement = false;
448             if (elementName.startsWith("-") && elementName.length() > 1) {
449                 elementName = elementName.substring(1);
450                 removeElement = true;
451             }
452             
453             ElementMetaData element = findElementAddPseudos(elementName);
454             if (removeElement) {
455                 elements.remove(element);
456             }
457             else {
458                 List<Parameter> elementParams = value.getParameters();
459                 element.applyBeanProps(elementParams);
460                 if (element.getOrder() == ElementMetaData.DEFAULT_ORDER) {
461                     element.setOrder(order++);
462                 }
463                 
464                 if (tabId != null) {
465                     element.setTabId(tabId);
466                 }
467             }
468         }
469     }

471     /**
472      * Applies a Bean's "actions" by adding ElementMetaData.
473      *
474      * @param values
475      */
476     private void applyActions(List<ParameterValue> values)
477     {
478         // Add action to the list of elements
479         for (ParameterValue value : values) {
480             String elementName = value.getValue();
481             boolean removeElement = false;
482             if (elementName.startsWith("-") && elementName.length() > 1) {
483                 elementName = elementName.substring(1);
484                 removeElement = true;
485             }
486             
487             String actionName = ACTION_PROPERTY_PREFIX + elementName;
488             ElementMetaData element = findElement(actionName);
489             if (removeElement) {
490                 if (element == null) {
491                     throw new RuntimeException("Action " + actionName + " does not exist in exposed list of actions.");
492                 }
493                 
494                 elements.remove(element);
495             }
496             else {
497                 if (element == null) {
498                     element = new ElementMetaData(this, actionName, createLabel(elementName), null);
499                     element.setAction(true);
500                     elements.add(element);
501                 }
502                 
503                 List<Parameter> elementParams = value.getParameters();
504                 element.applyBeanProps(elementParams);
505             }
506         }
507     }

509     /**
510      * Applies a Bean's "tabs" by adding ElementMetaData.
511      *
512      * @param values
513      */
514     private void applyTabs(List<ParameterValue> values)
515     {
516         // Add tab to the list of tabs
517         for (ParameterValue value : values) {
518             String tabName = value.getValue();
519             boolean removeTab = false;
520             if (tabName.startsWith("-") && tabName.length() > 1) {
521                 tabName = tabName.substring(1);
522                 removeTab = true;
523             }
524             
525             TabMetaData foundTab = null;
526             for (TabMetaData tab : tabs) {
527                 if (tab.getId().equals(tabName)) {
528                     foundTab = tab;
529                     break;
530                 }
531             }
532             
533             if (removeTab) {
534                 if (foundTab == null) {
535                     throw new RuntimeException("Tab " + tabName + " does not exist in exposed list of tabs.");
536                 }
537                 
538                 tabs.remove(foundTab);
539             }
540             else {
541                 if (foundTab == null) {
542                     foundTab = new TabMetaData(this, tabName, createLabel(tabName));
543                     tabs.add(foundTab);
544                 }
545                 
546                 List<Parameter> tabParams = value.getParameters();
547                 foundTab.applyBeanProps(tabParams);
548             }
549         }
550     }

565         String baseName = getBaseClassName(beanClass); // Name without pkg but with parent of inner class

569             if (shortName.equals(beanName) || baseName.equals(beanName) || fullName.equals(beanName)) {
570                 String beanContext = bean.getContext();
571                 if ((context == null && beanContext == null) ||
572                     (context != null && context.equals(beanContext))) {
573                     return bean;
574                 }
575             }
576         }
577         
578         // Default context implicitly exists. Also, don't require the context to be 
579         // explicitly specified for child beans.
580         if (context == null || isChildBean) {
581             return new Bean("", null, null, Collections.EMPTY_LIST);
582         }
583         
584         throw new RuntimeException("Bean context [" + context + "] does not exist.");

634     /**
635      * Creates a human readable label from a Java identifier.
636      * 
637      * @param identifier the Java identifier.
638      * 
639      * @return the label.
640      */
641     private static String createLabel(String identifier)
642     {
643         // Check for a complex property.
644         int idx = identifier.lastIndexOf('.');
645         if (idx < 0) {
646             idx = identifier.lastIndexOf('$'); // Java nested classes.
647         }
648         
649         if (idx >= 0 && identifier.length() > 1) {
650             identifier = identifier.substring(idx + 1);
651         }
652 
653         if (identifier.length() == 0) {
654             return "";
655         }
656         
657         char[] chars = identifier.toCharArray();
658         StringBuffer buf = new StringBuffer(chars.length + 10);
659 
660         // Capitalize the first letter.
661         buf.append(Character.toUpperCase(chars[0]));
662         boolean lastLower = false;
663         for (int i = 1; i < chars.length; ++i) {
664             if (!Character.isLowerCase(chars[i])) {
665                 // Lower to upper case transition -- add space before it
666                 if (lastLower) {
667                     buf.append(' ');
668                 }
669             }
670 
671             buf.append(chars[i]);
672             lastLower = Character.isLowerCase(chars[i]) || Character.isDigit(chars[i]);
673         }
674 
675         return buf.toString();
676     }
