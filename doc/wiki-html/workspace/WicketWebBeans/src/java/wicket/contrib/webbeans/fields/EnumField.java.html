20 package wicket.contrib.webbeans.fields;
21 
22 import wicket.contrib.webbeans.model.ElementMetaData;
23 import wicket.contrib.webbeans.model.NonJavaEnum;
24 import wicket.markup.html.form.DropDownChoice;
25 import wicket.markup.html.panel.Fragment;
26 import wicket.model.IModel;
27 
28 import java.util.List;
29 
30 /**
31  * A field for enumerated types. Presents the values as a drop-down list.
32  * Accepts a parameter of "default" indicating the default choice if the current selection is null.
33  * 
34  * @author Dan Syrstad
35  */
36 abstract public class EnumField extends AbstractField
37 {
38     /**
39      * Construct a new EnumField.
40      *
41      * @param id the Wicket id for the editor.
42      * @param model the model.
43      * @param metaData the meta data for the property.
44      * @param viewOnly true if the component should be view-only.
45      * @param values a List of values to be selected from. The element's toString() is used to 
46      *  produce the value displayed to the user.
47      */
48     public EnumField(String id, IModel model, ElementMetaData metaData, boolean viewOnly, List values)
49     {
50         super(id, model, metaData, viewOnly);
51 
52         Fragment fragment;
53         if (viewOnly) {
54             fragment = new Fragment("frag", "viewer");
55             fragment.add(new LabelWithMinSize("component", model));
56         }
57         else {
58             fragment = new Fragment("frag", "editor");
59             DropDownChoice choice = new DropDownChoice("component", model, values);
60             // Always allow the null choice.
61             choice.setNullValid(true);
62             fragment.add(choice);
63 
64             // Handle default selection if no model is provided
65             if (choice.getModelObject() == null) {
66                 String defaultChoice = metaData.getParameter("default");
67                 if (defaultChoice != null && values.size() > 0)
68                     setupDefault(values, defaultChoice, choice);
69             }
70         }
71 
72         add(fragment);
73     }
74 
75     private void setupDefault(List values, String defaultChoice, DropDownChoice choice)
76     {
77         boolean isJavaEnum = false;
78         Object firstValue = values.get(0);
79 
80         // Figure out what type of enum were dealing with
81         if (firstValue instanceof Enum) {
82             isJavaEnum = true;
83         }
84         else if (!(firstValue instanceof NonJavaEnum)) {
85             throw new RuntimeException("Unexpected enum type. Enum must be a Java language Enum or a custom enumeration that implements NonJavaEnum");
86         }
87 
88         // Iterate over the values and find the one that matches the default choice
89         for (Object value : values) {
90             if ((isJavaEnum && ((Enum)value).name().equals(defaultChoice)) ||
91                 (!isJavaEnum && ((NonJavaEnum)value).name().equals(defaultChoice)))  {
92                 choice.setModelObject(value);
93                 break;
94             }
95         }
96     }
97 }
