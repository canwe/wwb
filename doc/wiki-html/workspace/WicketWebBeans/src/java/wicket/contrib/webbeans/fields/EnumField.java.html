04 package wicket.contrib.webbeans.fields;
05 
06 import wicket.contrib.webbeans.model.ElementMetaData;
07 import wicket.contrib.webbeans.model.NonJavaEnum;
08 import wicket.markup.html.form.DropDownChoice;
09 import wicket.markup.html.panel.Fragment;
10 import wicket.model.IModel;
11 
12 import java.util.List;
13 
14 /**
15  * A field for enumerated types. Presents the values as a drop-down list.
16  * Accepts a parameter of "default" indicating the default choice if the current selection is null.
17  * 
18  * @author Dan Syrstad
19  */
20 abstract public class EnumField extends AbstractField
21 {
22     /**
23      * Construct a new EnumField.
24      *
25      * @param id the Wicket id for the editor.
26      * @param model the model.
27      * @param metaData the meta data for the property.
28      * @param viewOnly true if the component should be view-only.
29      * @param values a List of values to be selected from. The element's toString() is used to 
30      *  produce the value displayed to the user.
31      */
32     public EnumField(String id, IModel model, ElementMetaData metaData, boolean viewOnly, List values)
33     {
34         super(id, model, metaData, viewOnly);
35 
36         Fragment fragment;
37         if (viewOnly) {
38             fragment = new Fragment("frag", "viewer");
39             fragment.add(new LabelWithMinSize("component", model));
40         }
41         else {
42             fragment = new Fragment("frag", "editor");
43             DropDownChoice choice = new DropDownChoice("component", model, values);
44             // Always allow the null choice.
45             choice.setNullValid(true);
46             fragment.add(choice);
47 
48             // Handle default selection if no model is provided
49             if (choice.getModelObject() == null) {
50                 String defaultChoice = metaData.getParameter("default");
51                 if (defaultChoice != null && values.size() > 0)
52                     setupDefault(values, defaultChoice, choice);
53             }
54         }
55 
56         add(fragment);
57     }
58 
59     private void setupDefault(List values, String defaultChoice, DropDownChoice choice)
60     {
61         boolean isJavaEnum = false;
62         Object firstValue = values.get(0);
63 
64         // Figure out what type of enum were dealing with
65         if (firstValue instanceof Enum) {
66             isJavaEnum = true;
67         }
68         else if (!(firstValue instanceof NonJavaEnum)) {
69             throw new RuntimeException("Unexpected enum type. Enum must be a Java language Enum or a custom enumeration that implements NonJavaEnum");
70         }
71 
72         // Iterate over the values and find the one that matches the default choice
73         for (Object value : values) {
74             if ((isJavaEnum && ((Enum)value).name().equals(defaultChoice)) ||
75                 (!isJavaEnum && ((NonJavaEnum)value).name().equals(defaultChoice)))  {
76                 choice.setModelObject(value);
77                 break;
78             }
79         }
80     }
81 }
