020 package wicket.contrib.webbeans.fields;
021 
022 import wicket.contrib.webbeans.containers.BeanForm;
023 import wicket.contrib.webbeans.model.BeanPropertyModel;
024 import wicket.contrib.webbeans.model.ElementMetaData;
025 import wicket.markup.html.panel.Panel;
026 import wicket.model.IModel;
027 
028 
029 /**
030  * Base class for Fields. In general, Fields should <em>not</em> set renderBodyOnly to true
031  * if they want to be refreshed via Ajax.
032  *
033  * @author Dan Syrstad
034  */
035 abstract public class AbstractField extends Panel implements Field
036 {
037     private static final long serialVersionUID = -5452855853289381110L;
038     
039     private ElementMetaData elementMetaData;
040 
041     /**
042      * Construct a AbstractField. 
043      *
044      * @param id the Wicket id for the editor.
045      * @param model the model.
046      * @param metaData the meta data for the property.
047      * @param viewOnly true if the component should be view-only.
048      */
049     public AbstractField(String id, IModel model, ElementMetaData metaData, boolean viewOnly)
050     {
051         super(id, model);
052         this.elementMetaData = metaData;
053         
054         // Allow for refreshing of the field via Ajax.
055         setOutputMarkupId(true);
056         setRenderBodyOnly(false);
057     }
058     
059     /**
060      * {@inheritDoc}
061      * @see wicket.Component#onAttach()
062      */
063     @Override
064     protected void onAttach()
065     {
066         super.onAttach();
067         // If we're part of a BeanForm, register ourself with it.
068         BeanForm beanForm  = (BeanForm)findParent(BeanForm.class);
069         if (beanForm != null) {
070             Object bean = null;
071             if (getModel() instanceof BeanPropertyModel) {
072                 bean = ((BeanPropertyModel)getModel()).getBean();
073             }
074             
075             beanForm.registerComponent(this, bean, elementMetaData);
076         }
077     }
078 
079     /**
080      * Gets the elementMetaData.
081      *
082      * @return a ElementMetaData.
083      */
084     public ElementMetaData getElementMetaData()
085     {
086         return elementMetaData;
087     }
088 
089     /**
090      * Returns the ElementMetaData related to the specified propertyName.
091      * @param metaData metaData for this field.
092      * @param parameterName the parameter name on metaData. The value of this parameter should be the property name to lookup.
093      * @param propertyClass the class type the property should resolve to
094      * @return ElementMetaData
095      */
096     protected ElementMetaData getDependentProperty(ElementMetaData metaData, String parameterName, Class propertyClass) {
097         ElementMetaData property = null;
098         String propStr = metaData.getParameter(parameterName);
099         if (propStr != null) {
100             property = metaData.getBeanMetaData().findElement(propStr);
101             if (property == null) {
102                 throw new RuntimeException("'" + propStr + "' is not defined on " + metaData.getBeanMetaData().getBeanClass());
103             }
104 
105             if (!propertyClass.isAssignableFrom( property.getPropertyType() )) {
106                 throw new RuntimeException(parameterName + "'" + propStr + "' must return a " + propertyClass.getName() + " on "
107                                 + metaData.getBeanMetaData().getBeanClass() + ". Instead it returns "
108                                 + property.getPropertyType());
109             }
110         }
111         return property;
112     }
113 
114     /**
115      * Gets the dependent property bean from the specified property.
116      *
117      * @return the dependent property bean, or null if not defined.
118      * @param property the ElementMetaData property to extract the bean from
119      */
120     protected Object getDependentPropertyBean(ElementMetaData property)
121     {
122         if (property != null) {
123             BeanPropertyModel model = (BeanPropertyModel)getModel();
124             return property.getPropertyValue( model.getBean() );
125         }
126 
127         return null;
128     }
129 }
