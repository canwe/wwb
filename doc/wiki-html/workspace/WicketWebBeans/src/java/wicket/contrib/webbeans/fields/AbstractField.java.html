004 package wicket.contrib.webbeans.fields;
005 
006 import wicket.contrib.webbeans.containers.BeanForm;
007 import wicket.contrib.webbeans.model.BeanPropertyModel;
008 import wicket.contrib.webbeans.model.ElementMetaData;
009 import wicket.markup.html.panel.Panel;
010 import wicket.model.IModel;
011 
012 import java.util.Properties;
013 
014 
015 /**
016  * Base class for Fields. In general, Fields should <em>not</em> set renderBodyOnly to true
017  * if they want to be refreshed via Ajax.
018  *
019  * @author Dan Syrstad
020  */
021 abstract public class AbstractField extends Panel implements Field
022 {
023     private static final long serialVersionUID = -5452855853289381110L;
024     
025     private ElementMetaData elementMetaData;
026 
027     /**
028      * Construct a AbstractField. 
029      *
030      * @param id the Wicket id for the editor.
031      * @param model the model.
032      * @param metaData the meta data for the property.
033      * @param viewOnly true if the component should be view-only.
034      */
035     public AbstractField(String id, IModel model, ElementMetaData metaData, boolean viewOnly)
036     {
037         super(id, model);
038         this.elementMetaData = metaData;
039         
040         // Allow for refreshing of the field via Ajax.
041         setOutputMarkupId(true);
042         setRenderBodyOnly(false);
043     }
044     
045     /**
046      * {@inheritDoc}
047      * @see wicket.Component#onAttach()
048      */
049     @Override
050     protected void onAttach()
051     {
052         super.onAttach();
053         // If we're part of a BeanForm, register ourself with it.
054         BeanForm beanForm  = (BeanForm)findParent(BeanForm.class);
055         if (beanForm != null) {
056             Object bean = null;
057             if (getModel() instanceof BeanPropertyModel) {
058                 bean = ((BeanPropertyModel)getModel()).getBean();
059             }
060             
061             beanForm.registerComponent(this, bean, elementMetaData);
062         }
063     }
064 
065     /**
066      * Gets the elementMetaData.
067      *
068      * @return a ElementMetaData.
069      */
070     public ElementMetaData getElementMetaData()
071     {
072         return elementMetaData;
073     }
074 
075     /**
076      * Returns the ElementMetaData related to the specified propertyName.
077      * @param metaData metaData for this field.
078      * @param propertyName the propertyName to lookup
079      * @param propertyClass the class type the property should resolve to
080      * @return ElementMetaData
081      */
082     protected ElementMetaData getDependentProperty(ElementMetaData metaData, String propertyName, Class propertyClass) {
083         Properties config = metaData.getParameters();
084         ElementMetaData property = null;
085         String propStr = config.getProperty(propertyName);
086         if (propStr != null) {
087             property = metaData.getBeanMetaData().findElement(propStr);
088             if (property == null) {
089                 throw new RuntimeException("'" + propStr + "' is not defined on " + metaData.getBeanMetaData().getBeanClass());
090             }
091 
092             if (!propertyClass.isAssignableFrom( property.getPropertyType() )) {
093                 throw new RuntimeException(propertyName + "'" + propStr + "' must return a " + propertyClass.getName() + " on "
094                                 + metaData.getBeanMetaData().getBeanClass() + ". Instead it returns "
095                                 + property.getPropertyType());
096             }
097         }
098         return property;
099     }
100 
101     /**
102      * Gets the dependent property bean from the specified property.
103      *
104      * @return the dependent property bean, or null if not defined.
105      * @param property the ElementMetaData property to extract the bean from
106      */
107     protected Object getDependentPropertyBean(ElementMetaData property)
108     {
109         if (property != null) {
110             BeanPropertyModel model = (BeanPropertyModel)getModel();
111             return property.getPropertyValue( model.getBean() );
112         }
113 
114         return null;
115     }
116 }
